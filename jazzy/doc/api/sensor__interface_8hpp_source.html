174  virtual std::vector<StateInterface::ConstSharedPtr> on_export_state_interfaces()
175  {
176  // import the unlisted interfaces
177  std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
178  export_unlisted_state_interface_descriptions();
179 
180  std::vector<StateInterface::ConstSharedPtr> state_interfaces;
181  state_interfaces.reserve(
182  unlisted_interface_descriptions.size() + sensor_state_interfaces_.size());
183 
184  // add InterfaceDescriptions and create StateInterfaces from the descriptions and add to maps.
185  for (const auto & description : unlisted_interface_descriptions)
186  {
187  auto name = description.get_name();
188  unlisted_state_interfaces_.insert(std::make_pair(name, description));
189  auto state_interface = std::make_shared<StateInterface>(description);
190  sensor_states_map_.insert(std::make_pair(name, state_interface));
191  unlisted_states_.push_back(state_interface);
192  state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
193  }
194 
195  for (const auto & [name, descr] : sensor_state_interfaces_)
196  {
197  // TODO(Manuel) check for duplicates otherwise only the first appearance of "name" is inserted
198  auto state_interface = std::make_shared<StateInterface>(descr);
199  sensor_states_map_.insert(std::make_pair(name, state_interface));
200  sensor_states_.push_back(state_interface);
201  state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
202  }
203 
204  return state_interfaces;
205  }
