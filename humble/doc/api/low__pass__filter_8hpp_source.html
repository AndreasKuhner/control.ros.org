166  parameter_handler_.reset();

168  }
169  }
170  parameters_ = parameter_handler_->get_params();
171  compute_internal_params();

173  // Initialize storage Vectors

175  // TODO(destogl): make the size parameterizable and more intelligent is using complex types

184 template <>
185 inline bool LowPassFilter<geometry_msgs::msg::WrenchStamped>::update(
186  const geometry_msgs::msg::WrenchStamped & data_in, geometry_msgs::msg::WrenchStamped & data_out)
187 {
188  if (!this->configured_)
189  {
190  if (logger_)
191  RCLCPP_ERROR_SKIPFIRST_THROTTLE((*logger_), *clock_, 2000, "Filter is not configured");
192  return false;
193  }
194 
195  // Update internal parameters if required
196  if (parameter_handler_->is_old(parameters_))
197  {
198  parameters_ = parameter_handler_->get_params();
199  compute_internal_params();
200  }
201 
202  // IIR Filter
203  msg_filtered = b1_ * msg_old + a1_ * msg_filtered_old;
204  msg_filtered_old = msg_filtered;
205 
206  // TODO(destogl): use wrenchMsgToEigen
207  msg_old[0] = data_in.wrench.force.x;
208  msg_old[1] = data_in.wrench.force.y;
209  msg_old[2] = data_in.wrench.force.z;
210  msg_old[3] = data_in.wrench.torque.x;
211  msg_old[4] = data_in.wrench.torque.y;
212  msg_old[5] = data_in.wrench.torque.z;
213 
214  data_out.wrench.force.x = msg_filtered[0];
215  data_out.wrench.force.y = msg_filtered[1];
216  data_out.wrench.force.z = msg_filtered[2];
217  data_out.wrench.torque.x = msg_filtered[3];
218  data_out.wrench.torque.y = msg_filtered[4];
219  data_out.wrench.torque.z = msg_filtered[5];
220 
221  // copy the header
222  data_out.header = data_in.header;
223  return true;
224 }
