79 
80  v0 = v1 = static_cast<T>(0.0);

82  // Initialize the parameters once
83  if (!parameter_handler_)
84  {
85  try

89  this->param_prefix_);

91  catch (rclcpp::exceptions::ParameterUninitializedException & ex) {

98  parameter_handler_.reset();
99  return false;
100  }
101  }
102  parameters_ = parameter_handler_->get_params();
103  limiter = std::make_shared<control_toolbox::RateLimiter<T>>(
104  parameters_.min_value, parameters_.max_value,
105  parameters_.min_first_derivative_neg, parameters_.max_first_derivative_pos,
106  parameters_.min_first_derivative_pos, parameters_.max_first_derivative_neg,

113 template <typename T>
114 bool RateLimiter<T>::update(const T & data_in, T & data_out)
115 {
116  if (!this->configured_ || !limiter)
117  {
118  throw std::runtime_error("Filter is not configured");
119  }
120 
121  // Update internal parameters if required
122  if (parameter_handler_->is_old(parameters_))
123  {
124  parameters_ = parameter_handler_->get_params();
125  limiter->set_params(
126  parameters_.min_value, parameters_.max_value,
127  parameters_.min_first_derivative_neg, parameters_.max_first_derivative_pos,
128  parameters_.min_first_derivative_pos, parameters_.max_first_derivative_neg,
129  parameters_.min_second_derivative, parameters_.max_second_derivative
130  );
131  }
132  T v = data_in;
133  limiter->limit(v, v0, v1, static_cast<T>(parameters_.sampling_interval));
134  v1 = v0;
135  v0 = data_in;
136  data_out = v;
137  return true;
138 }
