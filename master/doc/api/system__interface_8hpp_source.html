224  std::vector<StateInterface::ConstSharedPtr> on_export_state_interfaces()
225  {
226  // import the unlisted interfaces
227  std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
228  export_unlisted_state_interface_descriptions();
229 
230  std::vector<StateInterface::ConstSharedPtr> state_interfaces;
231  state_interfaces.reserve(
232  unlisted_interface_descriptions.size() + joint_state_interfaces_.size() +
233  sensor_state_interfaces_.size() + gpio_state_interfaces_.size());
234 
235  // add InterfaceDescriptions and create the StateInterfaces from the descriptions and add to
236  // maps.
237  for (const auto & description : unlisted_interface_descriptions)
238  {
239  auto name = description.get_name();
240  unlisted_state_interfaces_.insert(std::make_pair(name, description));
241  auto state_interface = std::make_shared<StateInterface>(description);
242  system_states_.insert(std::make_pair(name, state_interface));
243  unlisted_states_.push_back(state_interface);
244  state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
245  }
246 
247  for (const auto & [name, descr] : joint_state_interfaces_)
248  {
249  auto state_interface = std::make_shared<StateInterface>(descr);
250  system_states_.insert(std::make_pair(name, state_interface));
251  joint_states_.push_back(state_interface);
252  state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
253  }
254  for (const auto & [name, descr] : sensor_state_interfaces_)
255  {
256  auto state_interface = std::make_shared<StateInterface>(descr);
257  system_states_.insert(std::make_pair(name, state_interface));
258  sensor_states_.push_back(state_interface);
259  state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
260  }
261  for (const auto & [name, descr] : gpio_state_interfaces_)
262  {
263  auto state_interface = std::make_shared<StateInterface>(descr);
264  system_states_.insert(std::make_pair(name, state_interface));
265  gpio_states_.push_back(state_interface);
266  state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
267  }
268  return state_interfaces;
