176  virtual std::vector<StateInterface::ConstSharedPtr> on_export_state_interfaces()
177  {
178  // import the unlisted interfaces
179  std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
180  export_unlisted_state_interface_descriptions();
181 
182  std::vector<StateInterface::ConstSharedPtr> state_interfaces;
183  state_interfaces.reserve(
184  unlisted_interface_descriptions.size() + joint_state_interfaces_.size());
185 
186  // add InterfaceDescriptions and create the StateInterfaces from the descriptions and add to
187  // maps.
188  for (const auto & description : unlisted_interface_descriptions)
189  {
190  auto name = description.get_name();
191  unlisted_state_interfaces_.insert(std::make_pair(name, description));
192  auto state_interface = std::make_shared<StateInterface>(description);
193  actuator_states_.insert(std::make_pair(name, state_interface));
194  unlisted_states_.push_back(state_interface);
195  state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
196  }
197 
198  for (const auto & [name, descr] : joint_state_interfaces_)
199  {
200  auto state_interface = std::make_shared<StateInterface>(descr);
201  actuator_states_.insert(std::make_pair(name, state_interface));
202  joint_states_.push_back(state_interface);
203  state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
204  }
205  return state_interfaces;
206  }
