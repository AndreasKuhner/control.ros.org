204  virtual std::vector<StateInterface::SharedPtr> on_export_state_interfaces()
205  {
206  // import the unlisted interfaces
207  std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
208  export_unlisted_state_interface_descriptions();
209 
210  std::vector<StateInterface::SharedPtr> state_interfaces;
211  state_interfaces.reserve(
212  unlisted_interface_descriptions.size() + joint_state_interfaces_.size());
213 
214  // add InterfaceDescriptions and create the StateInterfaces from the descriptions and add to
215  // maps.
216  for (const auto & description : unlisted_interface_descriptions)
217  {
218  auto name = description.get_name();
219  unlisted_state_interfaces_.insert(std::make_pair(name, description));
220  auto state_interface = std::make_shared<StateInterface>(description);
221  actuator_states_.insert(std::make_pair(name, state_interface));
222  unlisted_states_.push_back(state_interface);
223  state_interfaces.push_back(state_interface);
224  }
225 
226  for (const auto & [name, descr] : joint_state_interfaces_)
227  {
228  auto state_interface = std::make_shared<StateInterface>(descr);
229  actuator_states_.insert(std::make_pair(name, state_interface));
230  joint_states_.push_back(state_interface);
231  state_interfaces.push_back(state_interface);
232  }
233  return state_interfaces;
234  }
